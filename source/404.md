<script>
$(function() {
  var path = window.location.pathname;
  path = path.replace(/(\.html|\.htm|\.css|\.js)/, '');
  path = path.replace(/\//g,' ');
  path = path.substr(1); // trim first space

  if (path === '404') return;

  $('#st-results-container').appendTo($('#results-404'));

  $('#st-search-input').val(path);
  $('#st-search-input').focus();
});
</script>
